\begin{frame}{Trabalho Realizado}

\textbf{Conceitos e implementação}

\begin{itemize}
    \item Modelos dinâmicos discretos podem ser descritos por uma função recursiva:
    \begin{equation}
    P_{n+1} = P_n(a-bP_n),
    \end{equation}
    \item Sendo b um termo não linear, que representa fatores limitantes externos. No trabalho realizado foi implementado a forma normalizada dessa função, descrita por:
    \begin{equation}
        x_{n+1} = rx_n(1-x_n),x\in[0,1],r\in[0,4]
    \end{equation}
\end{itemize}

\end{frame}

% =================================================
% SLIDE 3
% =================================================
\begin{frame}{Resultados e Conclusões}

\textbf{Resultados}

\begin{itemize}
    \item $r\in[0,3]$ a função converge, $r\in[3,1+\sqrt{6}]$ a função diverge mas se mantém entre 2 pontos, $r\in[1+\sqrt{6},4]$ o número de pontos fixos aumenta progressivamente (sistema caótico)

\end{itemize}
