\begin{frame}{Trabalho Realizado}

\textbf{Conceitos e implementação}

\begin{itemize}
    \item O voo do drone é dividido em três partes, cada um com suas
        equações
    \item Parâmetros desconhecidos: $a_0, v_1, t_2$.
    \item As equações são obtidas a partir de um pouco de cálculo e
        transformações
        \begin{itemize}
            \item Fase 1: \begin{equation}
                    v(t) = a_0 t, \quad
                    x(t) = \frac12 a_0 t^2, \quad
                    t_1 = \frac{v_1}{a_0}.
                \end{equation}
            \item Fase 2:
                \begin{equation}
                    v(t) = v_1, \quad
                    x(t) = v_1 \left( t - \frac{t_1}2 \right).
                \end{equation}
            \item Fase 3:
                \begin{equation}
                    v(t) = -a_0 (t - t_2) + v_1, \quad
                    x(t) = -\frac{a_0}2 (t-t_2)^2 + v_1 \left( t - \frac{t_1}2 \right), \quad
                    t_3 = t_1 + t_2.
                \end{equation}
        \end{itemize}
\end{itemize}

\end{frame}
